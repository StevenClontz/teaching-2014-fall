\question[10]
Find the radius of convergence for the series
$\ds \sum_{n=0}^{\infty} \frac{2n\left(x-1\right)^n}{5^{n+1}}$

\begin{checkboxes}
\choice $R=2$
\choice $R=\frac{1}{2}$

\choice $R=\frac{1}{5}$
\choice None of these.
